C# Integration Code

Copy this code to your C# application:

// Initialize KeyAuth API
public static KeyAuthAPI KeyAuthApp = new KeyAuthAPI(
    name: "EXM",                    // App name
    ownerid: "X3lapZ9vPT",         // Account ID
    version: "1.5"                  // Application version
);

// Login function
public async Task<AuthResponse> Login(string username, string password)
{
    return await KeyAuthApp.Login(username, password);
}

// Register function
public async Task<AuthResponse> Register(string username, string password, string licenseKey)
{
    return await KeyAuthApp.Register(username, password, licenseKey);
}
